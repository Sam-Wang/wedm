	\centering
		\begin{tabular}{| l | l | l | l | p{4cm} |} \hline

		Current Sensor & HLSR 16P & 5 V &  & \parbox[t]{4cm}{Response Time = 2.5 $\mu$s \\ BW=400kHz \\ Accuracy=1\%} \vspace{-1mm} \\ \hline
		OP-AMP & MCP6002 & 5 V & 100$\mu$A & \parbox[t]{4cm}{BW=1MHz} \\ \hline
